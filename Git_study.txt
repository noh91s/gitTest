> 로컬(local) -> remote(github Repository)
echo "# gitTest" >> README.md
git init  // git 초기화 (필수)
git add README.md // .git 추가 모든파일 (필수)
git commit -m "first commit(메세지)" // commit을 해야함
git branch -M main // 브런치 확인 기본 master
git remote add origin https://github.com/noh91s/gitTest.git // 저장소 주소
git push -u origin main // 최종 실행

git config --global user.name 이름
git config --global user.email 등록이메일
